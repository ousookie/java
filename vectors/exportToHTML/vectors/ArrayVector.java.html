11       public ArrayVector(int vectorSize, double[] values) throws VectorIndexOutOfBoundsException {
12           if (vectorSize != values.length) {
13               throw new VectorIndexOutOfBoundsException(" VectorIndexOutOfBoundsException");
14           }
15           this.vectorSize = vectorSize;
16           this.values = values;
17       }
18   
19       @Override
20       public double[] getValues() {
21           return values;
22       }
23   
24       @Override
25       public int getVectorSize() {
26           return vectorSize;
27       }
28   
29       @Override
30       public double fastPower(double base, int exp) {
31           if (exp == 0) return 1;
32           if (exp % 2 == 0) return fastPower(base * base, exp / 2);
33           return base * fastPower(base, exp - 1);
34       }
35   
36       @Override
37       public double euclideanNorm() {
38           double sum = 0;
39           for (double element : values) {
40               sum += fastPower(element, 2);
41           }
42           return sqrt(sum);
43       }
44   }
